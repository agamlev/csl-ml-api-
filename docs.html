// CSL Won Probability API

# POST https://csl-ml-api.onrender.com/predict

Endpoint

POST https://csl-ml-api.onrender.com/predict

Parameters

FieldTypeRange
customer_health_scoreinteger0 – 100
customer_planstringBasic / Pro / Premium / Enterprise
usage_volumeinteger0 – 2500
restricted_feature_attemptinteger0 – 6
feature_related_inquiryinteger0 – 5
csat_scorefloat1.0 – 5.0

Request

// Body — application/json
{
  "customer_health_score":       75,
  "customer_plan":               "Pro",
  "usage_volume":                650,
  "restricted_feature_attempt":  1,
  "feature_related_inquiry":     3,
  "csat_score":                  4.2
}

⚠ Note

// This API runs on a free tier server.
// The first request after a period of inactivity may take 50+ seconds.
// If you receive a timeout error — wait a few seconds and try again.

Response

{
  "won_probability_%":  87.3,
  "verdict":            "likely won"  // "at risk" if below 60%
}
